-- 查询每张表所包含的条目数
select relname, reltuples::bigint 
from pg_class 
where 	relname like 'org%' 
	and substr(relname, length(relname)-3, 4)<>'pkey' 
	and substr(relname, length(relname)-2, 3)<>'seq';
